  <div class="container">
    <h2>💻 Cyber Session Calculator</h2>
    <label>بداية الوقت:</label>
    <input type="time" id="startTime">

    <label>نهاية الوقت:</label>
    <input type="time" id="endTime">

    <label>سعر الساعة (جنيه):</label>
    <input type="number" id="rate" value="15">

    <label>البينج (ms):</label>
    <input type="number" id="ping" value="50">

    <button onclick="calculate()">احسب السعر</button>

    <div class="result" id="result"></div>
  </div>

  <script>
    function calculate() {
      const start = document.getElementById("startTime").value;
      const end = document.getElementById("endTime").value;
      const rate = parseFloat(document.getElementById("rate").value);
      const ping = parseInt(document.getElementById("ping").value);

      if (!start || !end) {
        alert("من فضلك اختر بداية ونهاية الوقت");
        return;
      }

      const startTime = new Date("1970-01-01T" + start + ":00");
      const endTime = new Date("1970-01-01T" + end + ":00");

      let duration = (endTime - startTime) / (1000 * 60 * 60); // ساعات
      if (duration < 0) duration += 24; // لو عبر نص الليل

      const price = (duration * rate).toFixed(2);

      document.getElementById("result").innerHTML = `
        ⏱️ المدة: ${duration.toFixed(2)} ساعة<br>
        💰 السعر: ${price} جنيه<br>
        📶 البينج: ${ping} ms
      `;
    }
  </script>
